FILE (GLOB_RECURSE project_SRCS "*.hpp" "*.cpp")
SET (project_HEADERS "${PROJECT_NAME}.hpp")
SET (project_LIB ${PROJECT_NAME})

ADD_LIBRARY (esapp STATIC ${project_SRCS})
SET_TARGET_PROPERTIES (esapp PROPERTIES VERSION "${APPLICATION_VERSION_MAJOR}.${APPLICATION_VERSION_MINOR}" OUTPUT_NAME ${project_LIB} CLEAN_DIRECT_OUTPUT 1)

INSTALL (TARGETS esapp DESTINATION lib)
FILE (COPY ${project_HEADERS} DESTINATION ${HEADER_OUTPUT_PATH})

SET_PROPERTY (TARGET esapp PROPERTY CXX_STANDARD 11)
SET_PROPERTY (TARGET esapp PROPERTY CXX_STANDARD_REQUIRED ON)
